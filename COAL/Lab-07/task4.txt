INCLUDE Irvine32.inc

.data
	varA DWORD ?
	varB DWORD ?
	msgInput BYTE "Enter Two Numbers:-", 0
	msgConditionTrue BYTE "Condition True", 0
	msgConditionFalse BYTE "Condition False", 0
.code
main PROC
	MOV EDX, OFFSET msgInput
	call WriteString
	call Crlf
	call ReadInt
	MOV varA, EAX
	call ReadInt
	MOV varB, EAX

	MOV EAX, varA
	MOV EBX, varB

	CMP EAX, EBX
	JNG conditionFalse
	CMP EAX, 100
	JNL conditionFalse
	MOV EDX, OFFSET msgConditionTrue
	call WriteString
	JMP done

	conditionFalse:
		MOV EDX, OFFSET msgConditionFalse
		call WriteString
		JMP done
done:
exit
main ENDP
END main
