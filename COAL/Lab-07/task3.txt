INCLUDE Irvine32.inc

.data
	arr DWORD 5 DUP(?)
	msgInput BYTE "Enter Num: ", 0
	msgNonZero BYTE "Non Zero Numbers are: ", 0
	howMany DWORD ?
.code
main PROC
	MOV ECX, 5
	MOV ESI, OFFSET arr
	LOOPing:
		MOV EDX, OFFSET msgInput
		call WriteString
		call ReadInt
		MOV [ESI], EAX
		ADD ESI, TYPE DWORD
		CMP EAX, 0
		LOOPNZ LOOPing

	CMP EAX, 0
	JNZ skipFix
	INC ECX

	skipFix:
		MOV EDX, OFFSET msgNonZero
		call Crlf
		call WriteString
		MOV howMany, 5
		SUB howMany, ECX
		MOV EAX, howMany
		call WriteDec
	
exit
main ENDP
END main
