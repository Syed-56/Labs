INCLUDE Irvine32.inc

.data

.code
main PROC
	MOV AL, 10101110b
	AND AL, 11110110b	;AL=10100110
	call DumpRegs
	call Crlf

	MOV AL, 10101110b
	OR  AL, 11110110b	;AL=11111110
	call DumpRegs
	call Crlf

	MOV AL, 10101110b
	XOR AL, 11110110b	;AL = 01011000
	call DumpRegs
	call Crlf

	MOV AL, 10101110b
	TEST AL, 11110110b	;same as AND but only update flag not register
	call DumpRegs
	call Crlf
	exit
main ENDP
END main
