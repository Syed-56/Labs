INCLUDE Irvine32.inc

.data
	
.code
MAIN proc
	;EQUAL case
	MOV EAX, 10
	MOV EBX, 10
	CMP EAX, EBX	;EAX-EBX=0, SF,CF,OF = 0 and ZF=1

	;Less than case
	MOV EAX, 10
	MOV EBX, 20
	CMP EAX, EBX	;EAX-EBX=-10 ZF,OF=0 and CF,SF=1

	;greater than case
	MOV EAX, 10
	MOV EBX, 5
	CMP EAX, EBX	;EAX-EBX=5 SF,ZF,CF all = 0
	exit
main ENDP
END main
