INCLUDE Irvine32.inc

.data
	num1 SBYTE -100
	num2 BYTE 200
	num3 BYTE ?

.code
main PROC
	;for better practice we should use movsx for SBYTE and movzx for BYTE
	movsx eax, num1
	call WriteInt
	call Crlf
	movzx eax, num2
	call WriteInt
	call Crlf
	movzx eax, num3	;num3 stores garbage
	call WriteInt
	call Crlf
	exit
main ENDP
END main
